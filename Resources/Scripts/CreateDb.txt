
CREATE TABLE dbo.Cliente
(
    Id INT IDENTITY(1,1) NOT NULL,    
    Nombre NVARCHAR(50) NOT NULL,
	SapId NVARCHAR(50) NULL,
	Alias NVARCHAR(50) NULL,
	Comentarios NVARCHAR(1000) NULL,
	ObservacionesVentanas NVARCHAR(1000) NULL,
	ObservacionesBalconeras NVARCHAR(1000) NULL,
	ObservacionesPuertas NVARCHAR(1000) NULL

    CONSTRAINT PK_Cliente
        PRIMARY KEY CLUSTERED (Id),

    CONSTRAINT UQ_Cliente_Nombre 
        UNIQUE (Nombre),
);
GO
